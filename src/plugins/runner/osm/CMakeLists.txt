)

        writers/OsmObjectAttributeWriter.cpp

   )

        translators/O5mWriter.cpp
        translators/OsmConverter.cpp
   )

  OsmParser.cpp
  OsmPlugin.cpp
  OsmRunner.cpp
  OsmNode.cpp

)

marble_add_plugin( OsmPlugin ${osm_SRCS} ${osm_writers_SRCS} ${osm_translators_SRCS} )

endif()

endif()
